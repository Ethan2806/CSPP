soluList = ["Lakers", "Golden State Warriors", "Luka Doncic", "Stephen Curry"]

        print("Correct! Score: ")
    else: 
        print("Incorrect! The correct answer was " + soluList[quesCount] + "\n")

    print(quesAmount)
    quesCount += 1
